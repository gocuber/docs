# 配置

# 路由方法
get() 路由设置

　　get() 方法三个参数：
　　第一个参数：访问url；
　　第二个参数：路由规则，可以是一个闭包函数，也可以指定到控制器；
　　第三个参数：通配符限定规则； 


// 简单闭包
Route::get('/', function(){
	echo 'hello';
});

// 指定到控制器
Route::get('hello', 'demo/hello');

// 通配符限定
Route::get('detail/{id}/{name}', 'demo/detail/?id={:1}&name={:2}', ['id'=>'[0-9]+', 'name'=>'[a-z]+']);
wildcard() 通配符

　　使用 wildcard() 方法定义常用的全局约束通配符，get() 方法直接使用这些通配符；


// 使用正则来设置通配符

Route::wildcard(array(
	'id'   => '[0-9]+',             // + 匹配 1次~多次
	'name' => '[a-z]+',
	'year' => '[0-9]{4}',
	'all'  => '.*',                 // * 匹配 0次~多次
	'd'    => '\/?',                // ? 匹配 0次~1次
));

Route::get('detail/{id}', 'article/detail/?id={:1}');

Route::get('{all}', function(){
	// 404
});

// ...
domain() 子域名路由组

　　domain() 方法用来给指定域名设置一组路由规则，这组路由规则只在当前指定域名下有效；例如给用户中心模块指定一个子域名 ucenter.gocuber.com


Route::domain('ucenter.gocuber.com', function(){
    Route::get('{all}', 'ucenter/{:1}');
});
# 闭包路由
简单闭包

// 简单闭包
Route::get('/', function(){
	echo 'hello';
});

// 调用视图
Route::get('/', function(){
	View::display('demo_welcome');
});

// 404页
Route::get('/', function(){
	View::display('page_404');
});
通配符闭包

Route::get('detail/{id}', function($p1){
	echo $p1;
});

Route::get('news/{class}', function($p1){
	View::display($p1 . 'list');
}, ['class'=>'code|guide|demo']);

Route::get('news/{month}/{day}', function($p1, $p2){
	echo $p1 . '-' . $p2;
}, ['month'=>'[0-9]{2}', 'day'=>'[0-9]{2}']);
# 控制器路由

　　控制器路由路由格式：'模块/控制器/动作/?参数=值&参数=值'
　　如果是默认模块则可以省略；如果是默认动作 index 同样也可以省略；

简单控制器路由
// 下面三条配置实际是同一个规则，只要访问 hello，就会分配到默认default模块 welcome控制器 index动作来处理

Route::get('hello', 'welcome/index');

Route::get('hello', 'default/welcome/index');

Route::get('hello', 'welcome');

// 指定参数
Route::get('download', 'article/detail/?code=download');
通配符控制器路由

Route::get('detail/{id}', 'article/detail/?id={:1}');

Route::get('article/{class}', 'article/list/?code={:1}', ['class'=>'code|guide|demo']);

Route::get('user/{abc}/{id}', 'user/list/?code={:1}&id={:2}', ['abc'=>'[a-z]+']);
# 子域名路由组

　　domain() 方法用来给指定域名设置一组路由规则，这组路由规则只在当前指定域名下有效；


Route::domain('ucenter.gocuber.com', function(){

	// 这里设置的一组规则只在当前域下有效

	Route::get('hello', 'ucenter/user/info');
	Route::get('edit', 'ucenter/user/edit');
	// ...

    Route::get('{all}', 'ucenter/{:1}');
});
# 默认路由与全拦截路由
默认路由

　　当url中没有包含任何路由信息，如首页，就会执行 '/' 指定的控制器及动作；


// 默认路由就是一个 '/' 默认值 'default/index/index'
Route::get('/', 'default/index/index');
全拦截路由

// 全部拦截，将这行放在 routes.php 开始
Route::get('{all}', function(){
	echo '停服维护中..';
});

// 放在 routes.php 结尾，就会拦截配置之外的全部，相当于全部访问都需要配置在 routes.php 中
Route::get('{all}', function(){
	// page_404
});
