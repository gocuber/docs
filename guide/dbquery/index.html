# 查询
name() 指定表名

　　使用 name() 方法指定数据库表名称；

DB::connect()->name('users');
DB::connect()->name('users')->get();
DB::connect()->name('users')->line();
// ...
get() 查询多条记录
$list = DB::connect()->name('users')->get('id,name');
print_r($list);

array(
	array('id'=>1,'name'=>'name1'),
	array('id'=>2,'name'=>'name2'),
	// ..
);

　　get() 方法与 select() 方法返回的结果一样；如下，返回的结果 $list 是一样的；

$list = DB::connect()->select("select id,name from users");
print_r($list);

array(
	array('id'=>1,'name'=>'name1'),
	array('id'=>2,'name'=>'name2'),
	// ..
);
hash() 查询多条记录

　　有时候你需要以某字段做为key返回；

$list = DB::connect()->name('users')->hash('id','name,date');
print_r($list);

array(
	'1' => array('name'=>'name1','date'=>'2010-10-10'),
	'2' => array('name'=>'name2','date'=>'2010-10-11'),
	// ..
);

//或者直接返回一个 key value 格式的hash数组
$list = DB::connect()->name('users')->hash('id','name');
print_r($list);

array(
	'1' => 'name1',
	'2' => 'name2',
	// ..
);
line() 查询单条记录
$line = DB::connect()->name('users')->line('id,name');
print_r($line);

array('id'=>1,'name'=>'name1');
val() 查询一个字段
$val = DB::connect()->name('users')->val('name');

echo $val; //name2
count() 总数 max() 最大 min() 最小 avg() 平均 sum() 求和

//count() 总数
$val = DB::connect()->name('users')->count();
//或
$val = DB::connect()->name('users')->val('count(*)');

//max() 最大
$val = DB::connect()->name('users')->max('id');
//或
$val = DB::connect()->name('users')->val('max(id)');

//min() 最小
$val = DB::connect()->name('users')->min('id');
//或
$val = DB::connect()->name('users')->val('min(id)');

//avg() 平均
$val = DB::connect()->name('users')->avg('id');
//或
$val = DB::connect()->name('users')->val('avg(id)');

//sum() 求和
$val = DB::connect()->name('users')->sum('price');
//或
$val = DB::connect()->name('users')->val('sum(price)');
where() where条件

　　where() 方法三种调用格式；如下：

　　一、字符串格式：


	// where id = 1 and status = 1
	$db->where("id = 1 and status = 1");

　　二、hash格式与数组格式：


	// hash格式 [ key=>value, key=>value, ...]

	// where id = 1
	$db->where([ 'id'=>1 ]);

	// where id in (1,2)
	$db->where([ 'id'=>[1,2] ]);

	// where id in (1,2) and status = 1
	$db->where([ 'id'=>[1,2], 'status'=>1 ]);


	// 数组格式 [ [], [], ...]

	// where id = 1
	$db->where([ ['id','=',1] ]);

	// where id > 1
	$db->where([ ['id','>',1] ]);

	// where id = 1
	$db->where([ ['id',1] ]); // " = " 可以省略

	// where id in (1,2)
	$db->where([ ['id','in',[1,2]] ]);

	// where id > 1 and status = 1
	$db->where([ ['id','>',1], ['status',1] ]);

	// where id > 1 and status = 1
	$db->where([ ['id','>',1], ['status'=>1] ]); // 可以写成 key=>value 格式

	// where id > 1 and status = 1
	$db->where([ ['id','>',1], 'status'=>1 ]);   // 又一个 key=>value 格式 KV格式可省略[]

　　三、组合嵌套格式：['and|or', [], [] ...]
　　'and|or' 可省略，省略默认为 and，如果省略'and|or'实际上就是上面第二种格式；
　　同样的上面第二种格式 加上 'and|or' 就是组合嵌套格式；


	// 格式 ['and|or', [], [] ...]

	$db->where([ 'and', 'id'=>1, 'status'=>1 ]); // id=1 and status=1
	$db->where([ 'and', ['id','=',1], ['status','=',1] ]); // id=1 and status=1

	$db->where([ 'and', ['id',1], ['status',1] ]); // id=1 and status=1
	$db->where([ 'and', 'id'=>[1,2,3], 'status'=>1 ]); // id in (1,2,3) and status=1
	
	$db->where([ 'or', 'id'=>1, ['status','>',1] ]); // id=1 or status>1

	// 嵌套

	// where (id = 1 or status = 1) and (a = 1 or b = 1)
	$db->where([ 'and', ['or',['id'=>1],['status'=>1]], ['or',['a'=>1],['b'=>1]] ]);

	// where id>1 and (status=1 or name='name1')
	$db->where(['and', ['id','>',1], ['or',['status'=>1],['name'=>'name1']]]);

　　相信你已经发现以上三种格式可以总结为一种：['and|or', 表达式1, 表达式2 ...]
　　'and|or' 可省略，省略默认为 and；
　　表达式可以是 字符串 或 key=>value 或 数组 []；


	// where a in (1,2) or b = 1
	$db->where([ 'or', ['a'=>[1,2]], ['b'=>1] ]); // key=>value

	// where a in (1,2) or b = 1
	$db->where([ 'or', 'a'=>[1,2], 'b'=>1 ]); // key=>value 可以省略 []
	
	// where a = 1 or b = 1
	$db->where([ 'or', ['a','=',1], ['b','=',1] ]); // 数组 []

	// where a = 1 or b = 1
	$db->where([ 'or', ['a',1], ['b',1] ]); // " = " 可以省略


	// 最后使用 get() line() val() 返回结果
	$line = $db->name('user')->where(['id'=>1, 'status'=>1]])->line();
	print_r($line);

　　需要注意的是直接传字符串的时候不会对字符进行转义；除非你能确保你的sql是安全的；

andWhere() 追加and条件

	// where id=1 and status=1
	$db->where(['id'=>1])->andWhere('status'=>1);

	// where id=1 and status=1 and name='abc' and username='user1'
	$db->where([ 'id'=>1, 'status'=>1 ])
		->andWhere(['name'=>'abc'])
		->andWhere(['username'=>'user1']);
orWhere() 追加条件 orWhere

	//where id=1 or status=1
	$db->where(['id'=>1])->orWhere(['status'=>1]);

	//where check=1 or status=1
	$db->where(['check'=>1])->orWhere(['status'=>1]);

　　where() andWhere() orWhere() 三个方法的参数是一样的；

groupBy() 分组
//group by cid
DB::connect()->name('users')->groupBy('cid')->get();

//group by cid,pid
DB::connect()->name('users')->groupBy('cid,pid')->get();
having() 子句
//group by cid having sum(price) > 1000
DB::connect()->name('users')->groupBy('cid')->having('sum(price) > 1000')->get();
orderBy() 排序
//order by id desc
DB::connect()->name('users')->orderBy('id desc')->get();

//order by status desc,id desc
DB::connect()->name('users')->orderBy('status desc,id desc')->get();
offset() 与 limit() limit语句
DB::connect()->name('users')->limit(10)->get();                //limit 10
DB::connect()->name('users')->offset(20)->limit(10)->get();    //limit 20,10

　　where() andWhere() orWhere() orderBy() groupBy() offset() 与 limit() 几个方法都是链式使用，不能放在最后；如下：

DB::connect()->name('users')->limit(10)->get();

DB::connect()->name('users')->where()->limit(10)->get();

DB::connect()->name('users')->where()->orderBy()->limit(10)->get();

DB::connect()->name('users')->where()->get()->limit(10);             //错误的写法

// ...
innerJoin() 联表查询
//select a.id,a.name,b.groupname from user as a inner join user_group as b on a.gid=b.gid
DB::connect()->name('users as a')->innerJoin('user_group as b on a.gid=b.gid')->get();
leftJoin() 联表查询
//select a.id,a.name,b.groupname from user as a left join user_group as b on a.gid=b.gid
DB::connect()->name('users as a')->leftJoin('user_group as b on a.gid=b.gid')->get();
rightJoin() 联表查询
//select a.id,a.name,b.groupname from user as a right join user_group as b on a.gid=b.gid
DB::connect()->name('users as a')->rightJoin('user_group as b on a.gid=b.gid')->get();
field() 查询字段
//查询 id name
DB::connect()->name('users')->field('id,name')->get();

//可以直接写到get() line()方法
DB::connect()->name('users')->get('id,name');
query() 执行sql查询

　　当需要处理大数据的时候，像 get() select() 这样的方法就不太合适了，因为它们会把所有数据都读取到内存上。为了保持较低的内存消耗，使用 query() fetch()
　　用法如下：

$db = DB::connect();
$query = $db->name('users')->field('id,name')->where(['status'=>1])->query();
for (; $value = $db->fetch($query); ) {
    // ...
}

　　query() 只会执行查询sql语句；并不会返回查询结果；如果想要得到查询结果，还需使用 fetch() 方法；

$array = array();
$query = DB::connect()->query("select a.id,a.name,b.groupname from
							users as a inner join user_group as b
							on a.gid=b.gid");
for(;$value = DB::connect()->fetch($query);)
{
	$array[] = $value;
}

print_r($array);

array(
	array('id'=>1,'name'=>'name1','groupname'=>'系统用户'),
	array('id'=>2,'name'=>'name2','groupname'=>'注册用户'),
	// ..
);
# 增加

　　使用 insert() 方法增加一条记录；与之前使用基础里介绍的使用方法略有不同；其实就是两种不同的使用方法；可以直接传sql语句；也可以传一个 key=>value 字段对值 的数组；如下：

//直接写sql
DB::connect()->insert("insert into users (name,status) values ('?','?')", array( 'abc',1 ));

//或者传一个数组
DB::connect()->name('users')->insert(array( 'name'=>'abc', 'status'=>1 ));

　　insert() 方法返回最后插入的自增主键id，如果当前表有主键的话；如果想要插入一条空记录，返回最后 lastid 可以直接使用 insertEmpty() 方法；如下：

$lastid = DB::connect()->name('users')->insertEmpty();
# 删除

　　同样的也是两种用法

//直接写sql
DB::connect()->delete("delete from users where id=?", array( 10 ));

//使用链式方法 where() limit() ...
DB::connect()->name('users')->where(array('id',10))->delete();
DB::connect()->name('users')->orderBy('id desc')->limit(1)->delete();
# 修改
//直接写sql
DB::connect()->update("update user set status=? where name='?'", array( 1,'abc' ));

//或者传一个数组
DB::connect()->name('users')->where(array('name','abc'))->update(array( 'status'=>1 ));

　　delete() 和 update() 方法返回影响行数；
