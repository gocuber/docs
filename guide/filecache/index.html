# 配置

　　配置文件 config.php


	$_G['filecache']['default'] = array(
		'dir'       => SYSTEM_DIR . 'data/filecache/default/', // 缓存目录
	    'is_subdir' => 1,                                      // 是否自动生成子集缓存目录 默认1是 0否
	);

	// 可以按不同的功能模块配置多个
	$_G['filecache']['user'] = array(
	    'dir'       => SYSTEM_DIR . 'data/filecache/user/',
	    'is_subdir' => 0,
	);

　　为了防止缓存目录下的文件数过多，is_subdir 为1时会自动生成三层子集目录，如下：
SYSTEM_DIR . 'data/filecache/default/1c/59/3d/'

# 缓存连接 Cache_File::connect()

　　使用 Cache_File::connect() 连接文件缓存；


	Cache_File::connect();
	Cache_File::connect('user');
# 文件缓存使用
set() 写缓存

Cache_File::connect()->set('key1', 'CubePHP', 3600);  // 缓存一小时
get() 获取

Cache_File::connect()->get('key1');  // CubePHP
del() 删除缓存

Cache_File::connect()->del('key1');
setMulti() 一次写多个

$cache = Cache_File::connect();

$items = array(
	'key1' => 'CubePHP1',
	'key2' => 'CubePHP2',
	'key3' => 'CubePHP3',
);

$cache->setMulti($items, 3600);  // 缓存一小时
getMulti() 获取多个

$data = $cache->getMulti(array('key1','key2','key3'));

print_r($data);
array(
	'key1' => 'CubePHP1',
	'key2' => 'CubePHP2',
	'key3' => 'CubePHP3',
);
delMulti() 删除多个

$cache->delMulti(array('key1','key2','key3'));

　　不推荐使用文件缓存；
