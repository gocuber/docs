# 介绍

　　Model 模型是应用程序中用于处理数据逻辑的部分；
　　通常模型负责在数据库中存取数据；
　　一般情况下，一个数据库表对应一个 model 类文件；

　　直接实例化或使用 DB::model() 来调用model类文件；

# 创建model

　　新建 User 类继承父类 Model；

　　代码如下： 


	class User extends Model                  // 继承Model类
	{

		protected $_key        = 'default';   // 数据库config key 默认default

		protected $_name       = 'uc_user';   // 表名称

		protected $_primarykey = 'id';        // 主键

		protected $_fields = [                // 字段
			'id',            // 用户id
			'username',      // 账号
			'password',      // 密码
			'status',        // 状态
			'mobile',        // 手机
			'email',         // email
			'name',          // 姓名
			'createtime',    // 创建时间
			'updatetime',    // 更新时间
		];

	}
# model的使用

　　实例化或使用 DB::model() 来调用model类文件；
　　代码： 


	print_r((new User())->getFields());

	print_r(DB::model('User')->getFields());

	$user = new User();

	$user->get();

	DB::model('User')->get();

	DB::model('User')->where()->line();

	DB::model('User')->delete();

	// ...
# setTableName() 设置分表

　　可以随意扩充方法；如定义 setTableName() 方法，来实现数据库分表；代码：


	class User extends Model
	{
		// ...

		public function setTableName()
		{
			// 根据具体的业务需求来定义分表

			// $tablename = 'log_' . date('Ymd');
			// $tablename = 'stat_' . date('Ymd');
			// $tablename = 'md5_' . substr(md5('abc'), -2);

			$this->_name = $tablename;
			return $this;
		}
	}
# 使用命名空间创建model

　　代码：


	namespace models;

	class User extends \Model
	{
		// ...
	}

	// ...
	DB::model('models\User')->getFields();

	// ...
	(new models\User())->getFields();

	// ...
	use models\User;

	(new User())->getFields();
